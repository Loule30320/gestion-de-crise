        function fillRoleSelect() {

            const select = document.getElementById('role-select');

            for (const key in ROLES) {

                const role = ROLES[key];

                const option = document.createElement('option');

                option.value = key;

                option.textContent = `${role.name} (${role.scope})`;

                select.appendChild(option);

            }

        }

 

        function startGame() {

            const roleKey = document.getElementById('role-select').value;

           

            if (!roleKey) {

                // Remplacement de alert() par une fonction modale simple

                alertModal("Erreur", "Veuillez sélectionner un rôle pour commencer.");

                return;

            }

 

            // 1. Initialisation de l'état

            const selectedRole = ROLES[roleKey];

            const randomCrisis = CRISES[Math.floor(Math.random() * CRISES.length)];

 

            gameState.role = selectedRole;

            gameState.crisis = randomCrisis;

            gameState.turn = 1;

            gameState.severity = MAX_SEVERITY;

            gameState.resources = MAX_RESOURCES / 2;

            gameState.log = [];

 

            // 2. Mise à jour de l'UI

            document.getElementById('start-screen').classList.add('hidden');

            document.getElementById('game-screen').classList.remove('hidden');

 

            document.getElementById('current-role').textContent = `${selectedRole.name}`;

            document.getElementById('current-crisis').textContent = randomCrisis.name;

           

            updateUI();

           

            // 3. Log initial

            logMessage(`--- Début de la Simulation ---`);

            logMessage(`Vous êtes le **${selectedRole.name}**.`);

            logMessage(`Crise déclarée : **${randomCrisis.name}**. ${randomCrisis.description}`);

            logMessage(`**Mission :** Ramener la Sévérité à 0 en moins de ${MAX_TURNS} tours.`);

        }

 

        function updateUI() {

            // Mettre à jour les indicateurs

            const sevPercent = Math.max(0, Math.min(100, (gameState.severity / MAX_SEVERITY) * 100));

            const resPercent = Math.max(0, Math.min(100, (gameState.resources / MAX_RESOURCES) * 100));

           

            document.getElementById('current-turn').textContent = `${gameState.turn} / ${MAX_TURNS}`;

 

            document.getElementById('severity-bar').style.width = `${sevPercent}%`;

            document.getElementById('severity-bar').classList.toggle('bg-red-500', sevPercent > 50);

            document.getElementById('severity-bar').classList.toggle('bg-yellow-500', sevPercent <= 50 && sevPercent > 20);

            document.getElementById('severity-bar').classList.toggle('bg-green-500', sevPercent <= 20);

            document.getElementById('severity-value').textContent = `${Math.round(gameState.severity)}%`;

 

            document.getElementById('resource-bar').style.width = `${resPercent}%`;

            document.getElementById('resource-bar').classList.toggle('bg-green-500', resPercent > 20);

            document.getElementById('resource-bar').classList.toggle('bg-red-500', resPercent <= 20);

            document.getElementById('resource-value').textContent = `${Math.round(gameState.resources)} Unités`;

 

            // Mettre à jour le panneau d'actions

            renderActionPanel();

           

            // Vérifier les conditions de fin de jeu

            checkGameEnd();

        }

 

        function renderActionPanel() {

            const panel = document.getElementById('action-panel');

            panel.innerHTML = '';

           

            const availableActions = gameState.role.actions;

 

            for (const actionKey of availableActions) {

                // S'assurer que l'action existe avant de tenter d'accéder à ses propriétés

                const action = ACTIONS[actionKey];

               

                if (!action) {

                    console.error(`Action non définie: ${actionKey}. Vérifiez la liste ACTIONS.`);

                    continue; // Passer à l'action suivante pour éviter l'erreur

                }

 

                const crisis = gameState.crisis;

                const isCritical = crisis.critical_actions.includes(actionKey);

                const isNeutral = crisis.neutral_actions.includes(actionKey);

               

                let impactClass = 'bg-gray-100 hover:bg-gray-200 text-gray-800';

                let effectIcon = 'ℹ️';

 

                if (isCritical) {

                    impactClass = 'bg-red-100 hover:bg-red-200 text-red-800 border-red-300';

                    effectIcon = '🚨';

                } else if (isNeutral) {

                    impactClass = 'bg-blue-100 hover:bg-blue-200 text-blue-800 border-blue-300';

                    effectIcon = '💬';

                } else {

                    // Actions génériques

                    impactClass = 'bg-teal-100 hover:bg-teal-200 text-teal-800 border-teal-300';

                    effectIcon = '🛠️';

                }

               

                // Vérifier si l'action peut être entreprise (ressources)

                const isAffordable = gameState.resources >= Math.abs(action.resources);

                const disabled = isAffordable ? '' : 'opacity-50 cursor-not-allowed';

                const disabledAttr = isAffordable ? '' : 'disabled';

 

 

                const button = document.createElement('button');

                button.className = `action-button p-4 rounded-lg text-left border-l-4 ${impactClass} ${disabled} transition duration-150 ease-in-out`;

                button.setAttribute('onclick', `${disabledAttr ? '' : `takeAction('${actionKey}')`}`);

               

                // Calculer l'impact perçu (un bonus si c'est critique pour la crise)

                const severityChange = action.severity + (isCritical ? 10 : 0) - (isNeutral ? 5 : 0);

                const resourceChange = action.resources;

               

                let severityDisplay = severityChange > 0 ? `+${severityChange}% Sévérité` : '';

                let resourceDisplay = resourceChange > 0 ? `+${resourceChange} R` : `${resourceChange} R`;

                if (severityChange > 0) severityDisplay = `Baisse: ${severityChange}%`;

               

                button.innerHTML = `

                    <p class="font-bold text-lg mb-1">${effectIcon} ${action.name}</p>

                    <p class="text-sm italic mb-2">${action.description}</p>

                    <div class="text-xs font-mono">

                        <span class="p-1 rounded bg-white text-green-700 font-semibold mr-2">Effet Sévérité: ${severityChange > 0 ? '+' : ''}${severityChange}</span>

                        <span class="p-1 rounded bg-white ${resourceChange < 0 ? 'text-red-600' : 'text-blue-600'} font-semibold">Coût/Gain Ressources: ${resourceDisplay}</span>

                        ${!isAffordable ? '<span class="ml-2 text-red-500 font-bold">Ressources insuffisantes!</span>' : ''}

                    </div>

                `;

               

                panel.appendChild(button);

            }

        }

 

        function takeAction(actionKey) {

            const action = ACTIONS[actionKey];

            const crisis = gameState.crisis;

            const isCritical = crisis.critical_actions.includes(actionKey);

           

            // Vérifier les ressources une dernière fois

            if (gameState.resources < Math.abs(action.resources)) {

                logMessage(`**Échec!** L'action "${action.name}" a échoué par manque de ressources. (${Math.abs(action.resources)} nécessaires)`);

                nextTurn(); // Le tour avance quand même

                return;

            }

 

            // Calcul de l'impact réel

            let severityImpact = action.severity;

            let resourceImpact = action.resources;

           

            // Bonus pour action critique

            if (isCritical) {

                severityImpact *= 1.5; // Plus efficace si c'est une action critique pour cette crise

                logMessage(`**Succès!** L'action "${action.name}" était **critique** pour cette crise et a eu un impact accru.`);

            } else {

                logMessage(`L'action "${action.name}" a été entreprise.`);

            }

 

            // Application des changements

            gameState.severity -= severityImpact;

            gameState.resources += resourceImpact;

 

            // Avancer le tour

            nextTurn();

        }

       

        function nextTurn() {

            // Impact de la crise non gérée

            const crisisImpact = gameState.crisis.severity_impact * (gameState.severity / MAX_SEVERITY);

            gameState.severity -= crisisImpact; // La crise s'aggrave si la sévérité est élevée

           

            logMessage(`--- Tour ${gameState.turn} terminé ---`);

            logMessage(`La crise a progressé, impactant la sévérité de ${Math.round(crisisImpact)} points.`);

 

            gameState.turn++;

 

            // Clamper les valeurs

            gameState.severity = Math.max(0, Math.min(MAX_SEVERITY, gameState.severity));

            gameState.resources = Math.max(0, Math.min(MAX_RESOURCES, gameState.resources));

 

            updateUI();

        }

 

        function checkGameEnd() {

            if (gameState.severity <= 0) {

                // Victoire

                endGame("Victoire!", `Félicitations, vous avez maîtrisé la crise "${gameState.crisis.name}" en ${gameState.turn - 1} tours. Votre gestion en tant que **${gameState.role.name}** a été exemplaire !`);

            } else if (gameState.turn > MAX_TURNS) {

                // Défaite (Temps écoulé)

                endGame("Défaite...", `Le temps est écoulé ! La crise "${gameState.crisis.name}" n'a pas pu être résolue. La sévérité est restée à ${Math.round(gameState.severity)}%.`);

            } else if (gameState.resources <= 0 && gameState.severity > 5) {

                // Défaite (Ressources)

                endGame("Défaite...", `Vous avez épuisé toutes vos ressources ! Le manque de moyens a rendu impossible la poursuite de la gestion de crise.`);

            }

        }

 

        function logMessage(message) {

            gameState.log.push(message);

           

            const logElement = document.getElementById('crisis-log');

           

            // Créer le nouvel élément de log

            const p = document.createElement('p');

            p.className = 'text-gray-700';

            p.innerHTML = `**[T${gameState.turn}]** ${message}`;

            logElement.prepend(p); // Ajouter en haut

           

            // S'assurer que le journal ne dépasse pas une certaine taille pour l'affichage

            while (logElement.children.length > 20) {

                logElement.removeChild(logElement.lastChild);

            }

        }

       

        function endGame(title, message) {

            document.getElementById('end-title').textContent = title;

            document.getElementById('end-message').textContent = message;

            document.getElementById('end-modal').classList.remove('hidden');

        }

        function alertModal(title, message) {

            document.getElementById('end-title').textContent = title;

            document.getElementById('end-message').textContent = message;

           

            // Temporarily change the button to close the modal

            const endButton = document.querySelector('#end-modal button');

            endButton.textContent = "Fermer";

                document.getElementById('end-modal').classList.add('hidden');

                endButton.textContent = "Rejouer";

                endButton.onclick = resetGame;

            };

           

            document.getElementById('end-modal').classList.remove('hidden');

        }

            document.getElementById('game-screen').classList.add('hidden');

            document.getElementById('start-screen').classList.remove('hidden');

           

            // Réinitialiser l'état

            gameState.role = null;

            gameState.crisis = null;

           

            // Nettoyer le log pour le prochain jeu

            document.getElementById('crisis-log').innerHTML = '';

           

            // Reset les options si nécessaire (pas nécessaire ici car statique)

        }
